#### 阶乘
```go
package main
import "fmt"

func factorial(n uint64) (result uint64) {
	if(n > 0) {
		result = n * factorial(n-1)
		return result
	}
	return 1
}

func main() {
	var j int = 15
	fmt.Printf("%d 的阶乘是 %d\n",j,factorial(uint64(j)))

}
```

#### 斐波那契数列
```go
package main
import "fmt"
func fibonacci(n int) int {
	if n < 2 {
		return n
	}
	return fibonacci(n - 2) + fibonacci(n - 1)
}

func main() {
	for i := 0; i < 10; i++{
		fmt.Printf("%d\t",fibonacci(i))
	}
}

```
